    <!-- ===== THREE.JS: ЗАГРУЗКА ВНЕШНЕЙ 3D МОДЕЛИ ===== -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/three@0.128.0/examples/js/controls/OrbitControls.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/three@0.128.0/examples/js/loaders/GLTFLoader.js"></script>
    
    <script>
        // === СЦЕНА, КАМЕРА, РЕНДЕРЕР ===
        const container = document.getElementById('canvas-container');
        const scene = new THREE.Scene();
        scene.background = new THREE.Color(0x87ceeb);

        const camera = new THREE.PerspectiveCamera(
            60, container.clientWidth / container.clientHeight, 0.1, 1000
        );
        camera.position.set(8, 6, 12);

        const renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(container.clientWidth, container.clientHeight);
        renderer.shadowMap.enabled = true;
        renderer.shadowMap.type = THREE.PCFSoftShadowMap;
        container.appendChild(renderer.domElement);

        // === УПРАВЛЕНИЕ ===
        const controls = new THREE.OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;
        controls.dampingFactor = 0.05;
        controls.target.set(0, 2, 0);

        // === ОСВЕЩЕНИЕ ===
        const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);
        scene.add(ambientLight);

        const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
        dirLight.position.set(10, 15, 10);
        dirLight.castShadow = true;
        dirLight.shadow.mapSize.width = 2048;
        dirLight.shadow.mapSize.height = 2048;
        scene.add(dirLight);

        const hemiLight = new THREE.HemisphereLight(0xffffff, 0x444444, 0.4);
        scene.add(hemiLight);

        // === ЗЕМЛЯ ===
        const groundGeo = new THREE.PlaneGeometry(50, 50);
        const groundMat = new THREE.MeshLambertMaterial({ color: 0x4a7c3a });
        const ground = new THREE.Mesh(groundGeo, groundMat);
        ground.rotation.x = -Math.PI / 2;
        ground.receiveShadow = true;
        scene.add(ground);

        // === ЗАГРУЗКА 3D МОДЕЛИ ===
        const loader = new THREE.GLTFLoader();
        const loadingElement = document.getElementById('loading');

        // Используем пример модели с CDN (для демонстрации)
        // Замените на путь к вашей модели: 'models/house.glb'
        const modelPath = 'models/house.glb';

        loader.load(
            modelPath,
            function (gltf) {
                const model = gltf.scene;
                
                // Настройка модели
                model.scale.set(2, 2, 2);
                model.position.set(0, 0, 0);
                model.rotation.y = Math.PI;
                
                // Включаем тени для всех мешей
                model.traverse((object) => {
                    if (object.isMesh) {
                        object.castShadow = true;
                        object.receiveShadow = true;
                        
                        // Улучшаем материалы
                        if (object.material) {
                            object.material.needsUpdate = true;
                        }
                    }
                });
                
                scene.add(model);
                
                // Скрываем индикатор загрузки
                loadingElement.style.display = 'none';
                
                console.log('3D-модель успешно загружена!');
            },
            function (xhr) {
                // Прогресс загрузки
                if (xhr.total > 0) {
                    const percent = (xhr.loaded / xhr.total * 100).toFixed(0);
                    loadingElement.textContent = `Загрузка: ${percent}%`;
                }
            },
            function (error) {
                // Ошибка загрузки
                console.error('Ошибка загрузки модели:', error);
                loadingElement.textContent = 'Ошибка загрузки модели. Попробуйте обновить страницу.';
                loadingElement.style.color = '#ff6b6b';
            }
        );

        // === АНИМАЦИЯ ===
        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }
        animate();

        // === АДАПТИВНОСТЬ ===
        window.addEventListener('resize', () => {
            camera.aspect = container.clientWidth / container.clientHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(container.clientWidth, container.clientHeight);
        });
    </script>
